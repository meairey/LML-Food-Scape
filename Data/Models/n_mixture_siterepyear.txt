model {

for(i in 1:sites) {
   
   # Binomial model
   
   N[i] ~ dpois(lambda)


   # Observation model

   for(j in 1:temp) {
      for(k in 1:replicates){

         C[i, k, j] ~ dbin(p, N[i])
	
      }
    }
 }

## Priors
lambda ~ dunif(0,10)
p ~ dbeta(1,1)
# Derived Variable 
Ntotal = sum(N)

}
